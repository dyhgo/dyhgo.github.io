#include<bits/stdc++.h>

using namespace std;
priority_queue<int> p[100005];
int a[100005];
int main(){
	int n,m;
	cin>>n>>m;
	for(int i=1;i<=n;i++) p[i].push(0);
	for(int i=1;i<=n;i++) cin>>a[i];
	while(m--){
		int t1,t2;
		cin>>t1>>t2;
		p[t1].push(a[t2]);
		p[t2].push(a[t1]);
	}
	int ans=0;
	for(int i=1;i<=n;i++){
		if(p[i].top()<a[i]) ans++;
	}
	cout<<ans<<endl;
	
	return  0;
}

ll ans;
int main(){
	cin>>n;

		}
	}
	
}
